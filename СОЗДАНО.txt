💰 НАСТРОЙКА ЦЕН
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Откройте .env и измените:

BASE_PRICE=100         Базовая цена (руб)
PRICE_PER_KM=25       За километр (руб)
MIN_PRICE=150         Минимальная цена (руб)

Формула: Цена = BASE_PRICE + (расстояние × PRICE_PER_KM)

Примеры:
  • 2 км:  100 + 2×25 = 150 руб
  • 5 км:  100 + 5×25 = 225 руб
  • 10 км: 100 + 10×25 = 350 руб
